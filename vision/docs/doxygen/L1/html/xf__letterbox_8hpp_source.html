103  unsigned short input_col_index=0;

107  XF_TNAME(DEPTH, NPC) read_word_init;
108  for (int pixel = 0, bit1 = 0; pixel < NPC; pixel++, bit1 += (PLANES * 8)) {
109 #pragma HLS unroll
110  for (int channel = 0, bit2 = 0; channel < PLANES; channel++, bit2 += 8) {
111 #pragma HLS unroll

113  }
114  }

116  XF_TNAME(DEPTH, NPC) read_word_tmp=0;
117  XF_TNAME(DEPTH, NPC) read_word=0;

128 #pragma HLS pipeline II=1

130  //fprintf(stderr,"\n**** rowcol=%d row=%d col=%d",row_col_index, row_index, col_index);

164  #pragma HLS array_partition variable=read_word_extract dim=0
165  #pragma HLS array_partition variable=read_word_extract_prev dim=0
166  for (int pixel = 0, bit1 = 0; pixel < NPC; pixel++, bit1 += (PLANES * 8)) {
167 #pragma HLS unroll
168  for (int channel = 0, bit2 = 0; channel < PLANES; channel++, bit2 += 8) {
169 #pragma HLS unroll
170  read_word_extract[pixel][channel] = read_word.range(bit1 + (bit2 + 7), bit1 + bit2);
171  read_word_extract_prev[pixel][channel] = read_word_prev.range(bit1 + (bit2 + 7), bit1 + bit2);
172  }
173  }
174 
175  bool out_write_en;
176  if(current_valid_elements < NPC && input_col_index_prev == (imgInput_ncpr-1))
177  out_write_en = pad_col_count_left<NPC && input_col_index<(imgInput_ncpr+1) && pad_row_count_top<=0 && (row_index-dy)<in_height;
178  else
179  out_write_en = input_read_en;

182  #pragma HLS array_partition variable=DDR_write_data dim=0
183  for(int pixel=0; pixel<NPC; pixel++)
184  {
185  for (int channel = 0; channel < PLANES; channel++)
186  {
187

190  if((pixel + current_valid_elements)<NPC)
191  DDR_write_data[pixel][channel] = read_word_extract_prev[pixel + current_valid_elements][channel];
192  else

194  }
195  else
196  {
197  DDR_write_data[pixel][channel]=INSERT_VAL;

200 
201  }
202  }

206  for (int pixel = 0, bit1 = 0; pixel < NPC; pixel++, bit1 += (PLANES * 8)) {
207  // clang-format off

209  // clang-format on

211  // clang-format off
212 #pragma HLS unroll

215  }

217  }
218  _dst.write(write_index++, out_pix);

220 
221  //Last
222 
223  if(pad_col_count_left>=NPC)
224  pad_col_count_left-=NPC;

227  pad_row_count_top--;

243  }
